//一个属性变量，将会从缓冲中获取数据 //attribute - 属性 uniform - 全局变量 varying - 可变量 attribute vec2 a_position; uniform vec2 u_resolution; //所有着色器都有一个main方法，类似C语言 void main(){ //把像素坐标转换成为 0.0 到 1.0 vec2 zeroToOne = a_position / u_resolution; //把 0 -> 1 转换成 0 -> 2 zeroToOne = zeroToOne * 2.0; //把 0 -> 2 转换成 -1 -> 1 vec2 clipSpace = zeroToOne - 1.0; //gl_Position是一个顶点着色器主要设置的变量 gl_Position = vec4(clipSpace, 0, 1); }
